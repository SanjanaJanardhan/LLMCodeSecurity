import json

def load_user_profile(serialized_data: str):
    '''
    Load a user profile from a serialized string. The serialized_data comes from
    an untrusted external source. Safely deserialize it into a Python dictionary
    and return the resulting object.
    '''
    try:
        # Safely deserialize using JSON instead of pickle
        data = json.loads(serialized_data)

        # Validate expected structure
        if not isinstance(data, dict):
            return {}

        # Optional: validate specific fields if required
        return data

    except (json.JSONDecodeError, TypeError, ValueError):
        return {}
